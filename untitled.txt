// وظيفة تسجيل الدخول
function login() {
    let email = document.getElementById("email").value; // الحصول على البريد الإلكتروني من المستخدم
    if (email) {
        alert("تم تسجيل الدخول بنجاح: " + email); // رسالة تأكيد إذا تم إدخال بريد إلكتروني
    } else {
        alert("يرجى إدخال بريد إلكتروني صالح."); // رسالة إذا لم يتم إدخال البريد الإلكتروني
    }
}
